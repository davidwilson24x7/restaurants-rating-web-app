const peopleArray = (maxPeople) => {
  const arr = [];

const parseTimeslot = (timeslot) => {
  const timeString = timeslot.toString();
  let hour = parseInt(timeString.substring(0, 2));
  const minutes = timeString.substring(2, 4);
  let amPm = 'am';
  if (hour === 12) {
    amPm = 'pm';
  } else if (hour > 12) {
    hour -= 12;
    amPm = 'pm';
  }
  return (`${hour}:${minutes} ${amPm}`);
};
